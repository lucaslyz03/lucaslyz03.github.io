1.数据的导入导出

  • 使用csv文件格式
#导入
def inputdata():    
    with open('datacenter.csv', newline='', encoding="utf-8") as csvfile:
        reader= csv.DictReader(csvfile)
        dataList=[]
        for row in reader:
            dataList.append(row)  #将每一个人的信息存为一个字典
    return dataList   #再将每一个字典存入列表
  • 可以看到，整个数据库为一个列表，表格中的每行对应列表的一个元素，每个人的信息存为一个键-值对，用字典存储
#导出
def outputdata(data):
    with open('datacenter.csv', 'w', newline='', encoding="utf-8") as FF:
        fieldnames = ['\ufeffid','classNum','name','score','div']
        writer = csv.DictWriter(FF, fieldnames=fieldnames)
        #write changes to the file
        writer.writeheader()
        for row in data:
            writer.writerow(row)
